## Iterable protocol

```javascript
const range = {
    from: 1,
    to: 5,
    [Symbol.iterator]() {
        let current = this.from;
        const last = this.to;
        return {
            next() {
                if (current <= last) {
                    return { value: current++, done: false };
                } else {
                    return { value: undefined, done: true };
                }
            }
        };
    }
};

for (const x of range) console.log(x);   // 1, 2, 3, 4, 5
[...range]                                 // [1, 2, 3, 4, 5]
```

## Generator

